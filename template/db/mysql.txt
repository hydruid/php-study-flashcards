create database phpstudyflashcards;
GRANT ALL PRIVILEGES ON phpstudyflashcards.* TO 'zabbix'@'localhost';
FLUSH PRIVILEGES;

CREATE TABLE projects (
  id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  project VARCHAR(100) NOT NULL,
  var1 VARCHAR(300) NOT NULL,
  var2 VARCHAR(300) NOT NULL,
  date TIMESTAMP
);

CREATE TABLE questions (
  id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  project VARCHAR(100) NOT NULL,
  question VARCHAR(500) NOT NULL,
  answer1 VARCHAR(300) NOT NULL,
  answer2 VARCHAR(300) NOT NULL,
  answer3 VARCHAR(300) NOT NULL,
  answer4 VARCHAR(300) NOT NULL,
  correct VARCHAR(300) NOT NULL,
  date TIMESTAMP
);

CREATE TABLE quiz (
  id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  questionid VARCHAR(100) NOT NULL,
  answer VARCHAR(300) NOT NULL,
  correct VARCHAR(300) NOT NULL,
  status VARCHAR(300) NOT NULL,
  date TIMESTAMP
);

CREATE TABLE notes (
  id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  note VARCHAR(300) NOT NULL,
  date TIMESTAMP
);
